```env
# === IA (OPCIONAL) ===

```

### 🐳 **Docker Compose Premium**
```yaml
version: '3.8'
services:
  waha-mcp-plus:
    build:
      dockerfile: Dockerfile.plus
    ports:
      - "8000:8000"    # MCP Server
      - "8080:8080"    # Dashboard
    environment:
      WAHA_BASE_URL: http://waha:3000
      WAHA_ENABLE_ANALYTICS: "true"
    depends_on:
      - waha
      - redis

  waha:
    image: devlikeapro/waha:latest
    ports:
      - "3000:3000"
    
  redis:
    image: redis:7-alpine
    ports:
      - "6379:6379"
```
